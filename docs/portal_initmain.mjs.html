let discordClient = null;
let shutdownInProgress = false;

/**
 * Handles termination signals and cleanly shuts down Discord bot.
 * @param {string} signal - OS signal string
 */
async function gracefulShutdown(signal) {
    if (shutdownInProgress) return;
    shutdownInProgress = true;

    printDivider();
    printStep(`‚ö†Ô∏è  Received ${signal}. Shutting down Noona-Portal...`);

    try {
        if (discordClient) {
            printStep('üßº Destroying Discord client...');
            await discordClient.destroy();
            printResult('‚úÖ Discord client shut down.');
        }

        printResult('üåô Cleanup complete.');
    } catch (err) {
        printError(`‚ùå Error during shutdown: ${err.message}`);
    } finally {
        printDivider();
        printResult('üëã Noona-Portal exited gracefully.');
        process.exit(0);
    }
}

process.on('SIGINT', () => gracefulShutdown('SIGINT'));
process.on('SIGTERM', () => gracefulShutdown('SIGTERM'));

(async () => {
    console.log('');
    printHeader('Noona-Portal');
    printSection('üõ†Ô∏è Initializing Portal Environment...');

    const summary = [];

    // 1. üì° Vault Availability Check
    printStep('üì° Checking Vault availability...');
    try {
        const vaultReady = await waitForVaultReady();
        if (!vaultReady) throw new Error('Timeout or network failure');
        printResult('‚úÖ Vault is online.');
        summary.push({ name: 'Vault Connection', info: 'Responding at /health', ready: true });
    } catch (err) {
        printError(`‚ùå Vault check failed: ${err.message}`);
        summary.push({ name: 'Vault Connection', info: err.message, ready: false });
    }

    // 2. üîê Initialize Auth (includes publicKey fetch + pair check)
    printStep('üîê Initializing Auth...');
    try {
        const authReady = await initAuth();
        if (!authReady) throw new Error('Auth setup failed');
        printResult('‚úÖ Auth system initialized');
        summary.push({ name: 'Auth', info: 'JWT keys verified and cached', ready: true });
    } catch (err) {
        printError(`‚ùå Auth init failed: ${err.message}`);
        summary.push({ name: 'Auth', info: err.message, ready: false });
    }

    // 3. üîë Vault Auth
    printStep('üîë Skipping Vault token logic (public-key-only system)');
    summary.push({ name: 'Vault Auth', info: 'üü° PublicKey Mode', ready: true });

    // 4. üìö Kavita Authentication
    printStep('üìö Authenticating with Kavita...');
    try {
        const success = await authenticateWithKavita();
        if (!success) throw new Error('Authentication failed');
        printResult('‚úÖ Kavita authentication successful.');
        summary.push({ name: 'Kavita API', info: 'Authenticated successfully', ready: true });
    } catch (err) {
        printError(`‚ùå Kavita auth failed: ${err.message}`);
        summary.push({ name: 'Kavita API', info: err.message, ready: false });
    }

    // 5. ü§ñ Discord Bot Startup
    printStep('ü§ñ Starting Discord bot...');
    try {
        const result = await setupDiscord();
        if (result.client) {
            discordClient = result.client;
            summary.push({
                name: 'Discord Bot',
                info: `Client logged in (${result.commandCount} commands)`,
                ready: true
            });
        } else {
            summary.push({
                name: 'Discord Bot',
                info: '‚ö†Ô∏è Discord token invalid ‚Äî bot disabled',
                ready: false
            });
        }
    } catch (err) {
        printError(`‚ùå Discord bot init failed: ${err.message}`);
        summary.push({ name: 'Discord Bot', info: err.message, ready: false });
    }

    // 6. üîî Notification Scheduler
    printStep('üîî Initializing notification service...');
    try {
        if (!discordClient) throw new Error('Skipped (no Discord client)');
        await setupLibraryNotifications(discordClient);
        const interval = process.env.CHECK_INTERVAL_HOURS || '168';
        printDebug(`CHECK_INTERVAL_HOURS=${interval} [NODE_ENV: ${process.env.NODE_ENV}]`);
        summary.push({
            name: 'Library Notifier',
            info: `Initialized (interval: ${interval}hr)`,
            ready: true
        });
    } catch (err) {
        printError(`‚ùå Library notifier failed: ${err.message}`);
        summary.push({ name: 'Library Notifier', info: err.message, ready: false });
    }

    // ‚úÖ Boot Summary Table
    printBootSummary(summary);
})();
